#include <stdio.h>

int main() {
    int millis=45000;
    int secondes=0;
    int minutes=0;
    int heures=0;
    while(millis >= 1000){
        millis-=1000;
        secondes++;
        if(secondes==60){
            secondes-=60;
            minutes++;
        }
        if(minutes==60){
            minutes-=60;
            heures++;
        }
    }
    printf("%i\n",millis);
    printf("%i\n",secondes);
    printf("%i\n",minutes);
    printf("%i\n",heures);
    printf("Le temps total est de %i",heures," heures ",minutes, "minutes ",secondes," secondes ",millis," milli-secondes");
}
